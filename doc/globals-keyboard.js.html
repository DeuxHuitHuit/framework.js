	global.keyFromCode = function (code) {
		var key = '?';
		if (!code) {
			return key;
		}
		$.each(window.keys, function (index, value) {
			if (code === value) {
				key = index;
				return false;
			}
			
			return true;
		});
		return key;
	};
	
	// Chars
	global.isChar = function (c) {
		return c === window.keys.space_bar || (c > window.keys['0'] && c <= window.keys.z);
	};
